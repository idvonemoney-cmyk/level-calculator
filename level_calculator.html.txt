  <script>
    // 升級所需經驗公式
    function xpRequired(n) {
      if (n === 1) return 100; // 0→1 特殊處理
      return 100 + (((n - 2) * 10 + (55 * 2)) * ((n - 1) / 2));
    }

    // 計算單一等級
    function calculate() {
      const level = parseInt(document.getElementById("level").value);
      const resultDiv = document.getElementById("result");

      if (isNaN(level) || level < 1 || level > 100) {
        resultDiv.textContent = "⚠️ 請輸入 1 到 100 之間的等級";
        return;
      }

      let totalXP = 0;
      for (let i = 1; i <= level; i++) {
        totalXP += xpRequired(i);
      }
      const minutes = Math.ceil(totalXP / 20);
      const hours = (minutes / 60).toFixed(1);

      resultDiv.innerHTML = `
        🎯 目標等級：<b>${level} 級</b><br>
        📈 總共需要：<b>${totalXP.toLocaleString()} XP</b><br>
        ⏳ 預估時間：約 <b>${minutes.toLocaleString()} 分鐘</b> (${hours} 小時)
      `;
    }

    // 生成等級對照表
    function generateTable() {
      const tableBody = document.getElementById("levelTable");
      tableBody.innerHTML = "";
      let cumulativeXP = 0;
      for (let level = 1; level <= 100; level++) {
        const needXP = xpRequired(level);
        cumulativeXP += needXP;
        const minutes = Math.ceil(cumulativeXP / 20);
        const hours = (minutes / 60).toFixed(1);
        const row = `
          <tr>
            <td>${level}</td>
            <td>${needXP.toLocaleString()}</td>
            <td>${cumulativeXP.toLocaleString()}</td>
            <td>${minutes.toLocaleString()} 分鐘 (${hours} 小時)</td>
          </tr>
        `;
        tableBody.innerHTML += row;
      }
    }
    generateTable();

    // 分頁切換
    function switchTab(tabId) {
      document.querySelectorAll(".tab").forEach(tab => tab.classList.remove("active"));
      document.querySelectorAll(".tabBtn").forEach(btn => btn.classList.remove("active"));
      document.getElementById(tabId).classList.add("active");
      event.target.classList.add("active");
    }
